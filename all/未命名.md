    end

    subgraph Business_Layer [业务层: Spring Boot]
        Auth_Service[用户认证服务]

        Traffic_Service[客流分析服务]
        Profile_Service[游客画像服务]
        Recommend_Service[智能推荐服务]

    end

        MySQL[("MySQL 关系型数据库")]

    end

    Presentation_Layer <--> Interface_Layer
    Interface_Layer <--> Business_Layer
    Business_Layer <--> Data_Layer
